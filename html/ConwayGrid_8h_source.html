58 bool tick(){
59 Store temp = storage;
60 auto evolved = false;
61 for(auto x = storage.min_x() - 1; x < storage.max_x() + 1; x++){
62 for(auto y = storage.min_y() - 1; y < storage.max_y() + 1; y++){
63 auto n = getAliveNeighbors({x, y});
64 auto wasAlive = isAlive({x, y});
65 auto alive = rules(wasAlive, n);
66 if(alive != wasAlive){
67 evolved = true;
68 if(alive){
69 temp.set(x, y);
70 }else{
71 temp.remove(x, y);
72 }
73 }
74 }
75 }
76 storage = temp;
77 return evolved;
78 }
